```ts
import { readFileSync } from 'node:fs/promises';
import { parse } from 'yaml';
import type { Skill, SkillFrontmatter } from './types.js';

export function parseSkillMarkdown(content: string, filePath: string): Skill {
  // YAML frontmatter between --- markers
  const fmMatch = content.match(/^---\n([\s\S]+?)\n---\n([\s\S]+)$/);
  if (!fmMatch) {
    throw new Error(`Skill ${filePath} missing frontmatter`);
  }
  const frontmatter = parse(fmMatch[1]!) as SkillFrontmatter;
  const body = fmMatch[2]!;
  return {
    name: frontmatter.name,
    description: frontmatter.description,
    frontmatter,
    body,
    sourcePath: 'project', // set by caller
    filePath,
  };
}
```

## Task 3: Skill loader with precedence

**Files:**

- `packages/workflow/src/loader.ts` (complete)

**Precedence (Q8 grill):** project `.awecode/skills/` > user `~/.config/awecode/skills/` > built-in `awecode/skills/`

```ts
import { stat } from 'node:fs/promises';
import { join } from 'node:path';
import { homedir } from 'node:os';
import { fileURLToPath } from 'node:url';

const BUILT_IN_DIR = fileURLToPath(new URL('../../awecode/skills/', import.meta.url));

export async function loadSkill(name: string, projectRoot: string): Promise<Skill | null> {
  const candidates = [
    { dir: join(projectRoot, '.awecode', 'skills'), source: 'project' as const },
    { dir: join(homedir(), '.config', 'awecode', 'skills'), source: 'user' as const },
    { dir: BUILT_IN_DIR, source: 'built-in' as const },
  ];

  for (const candidate of candidates) {
    const filePath = join(candidate.dir, name, 'SKILL.md');
    try {
      await stat(filePath);
      const content = await readFileSync(filePath, 'utf-8');
      const skill = parseSkillMarkdown(content, filePath);
      skill.sourcePath = candidate.source;
      return skill;
    } catch (err) {
      if ((err as NodeJS.ErrnoException).code !== 'ENOENT') throw err;
    }
  }
  return null;
}

export async function listAvailableSkills(projectRoot: string): Promise<string[]> {
  const names = new Set<string>();
  const candidates = [
    join(projectRoot, '.awecode', 'skills'),
    join(homedir(), '.config', 'awecode', 'skills'),
    BUILT_IN_DIR,
  ];
  for (const dir of candidates) {
    try {
      const entries = await readdir(dir, { withFileTypes: true });
      for (const e of entries) {
        if (e.isDirectory()) names.add(e.name);
      }
    } catch { /* dir doesn't exist, skip */ }
  }
  return [...names];
}
```

```ts
  args: Record<string, unknown>,

    skillName: name,
    output: skill.body,
  };
}

export async function startWorkflow(
  name: string,
  projectRoot: string,
): Promise<{ ok: true; skillBody: string } | { ok: false; error: string }> {
  try {
    const skill = await loadSkill(name, projectRoot);
    if (!skill) {
      return { ok: false, error: `Workflow "${name}" not found` };
    }
    return { ok: true, skillBody: skill.body };
  } catch (err) {
    // Q23 grill: fail-loud, agent fallback to Direct Mode
    return { ok: false, error: (err as Error).message };
  }
}
```

```ts
// packages/repomap/src/parser.ts
import { Parser, Language } from 'web-tree-sitter';
import path from 'node:path';

let parserInstance: Parser | null = null;
const languageCache = new Map<string, Language>();

async function getParser(): Promise<Parser> {
  if (!parserInstance) {
    await Parser.init();
    parserInstance = new Parser();
  }
  return parserInstance;
}

async function getLanguage(ext: string): Promise<Language | null> {
  if (languageCache.has(ext)) return languageCache.get(ext)!;
  const langMap: Record<string, string> = {
    '.ts': 'tree-sitter-typescript/typescript',
    '.tsx': 'tree-sitter-typescript/tsx',
    '.js': 'tree-sitter-javascript',
    '.jsx': 'tree-sitter-javascript',
    '.py': 'tree-sitter-python',
    '.go': 'tree-sitter-go',
    '.rs': 'tree-sitter-rust',
  };
  const langPath = langMap[ext];
  if (!langPath) return null;
  const lang = await Language.load(langPath);
  languageCache.set(ext, lang);
  return lang;
}

export interface ParsedSymbol {
  name: string;
  kind: 'function' | 'class' | 'method' | 'variable';
  signature: string;
  startLine: number;
}

export async function parseFile(filePath: string, content: string): Promise<{ symbols: ParsedSymbol[] } | { unsupported: true }> {
  const ext = path.extname(filePath);
  const lang = await getLanguage(ext);
  if (!lang) return { unsupported: true };

  const parser = await getParser();
  parser.setLanguage(lang);
  const tree = parser.parse(content);

  const symbols: ParsedSymbol[] = [];
  walkTree(tree!.rootNode, symbols);
  return { symbols };
}

function walkTree(node: any, symbols: ParsedSymbol[]): void {
  const symbolKinds = ['function_declaration', 'class_declaration', 'method_definition'];
  if (symbolKinds.includes(node.type)) {
    symbols.push({
      name: node.childForFieldName('name')?.text ?? '<anon>',
      kind: node.type.includes('class') ? 'class' : node.type.includes('method') ? 'method' : 'function',
      signature: node.text.slice(0, node.text.indexOf('{')),
      startLine: node.startPosition.row + 1,
    });
  }
  for (let i = 0; i < node.childCount; i++) {
    walkTree(node.child(i)!, symbols);
  }
}
```

## Task 13: Repo Map cache by commit hash

```ts
import { readFile, writeFile, mkdir } from 'node:fs/promises';
import { join, dirname } from 'node:path';
import { simpleGit } from 'simple-git';
import type { RankedFile } from './ranker.js';

export async function getCommitHash(projectRoot: string): Promise<string> {
  const git = simpleGit(projectRoot);
  return (await git.revparse(['HEAD'])).trim();
}

export function getCachePath(projectRoot: string): string {
  return join(projectRoot, '.awecode', 'cache', 'repo-map.json');
}

export async function loadCachedMap(projectRoot: string): Promise<{ commitHash: string; files: RankedFile[] } | null> {
  try {
    const content = await readFile(getCachePath(projectRoot), 'utf-8');
    return JSON.parse(content);
  } catch { return null; }
}

export async function saveCachedMap(projectRoot: string, data: { commitHash: string; files: RankedFile[] }): Promise<void> {
  const path = getCachePath(projectRoot);
  await mkdir(dirname(path), { recursive: true });
  await writeFile(path, JSON.stringify(data, null, 2), 'utf-8');
}

export async function getOrGenerateMap(projectRoot: string, generator: () => Promise<RankedFile[]>): Promise<RankedFile[]> {
  const currentHash = await getCommitHash(projectRoot);
  const cached = await loadCachedMap(projectRoot);
  if (cached && cached.commitHash === currentHash) {
    return cached.files;
  }
  const fresh = await generator();
  await saveCachedMap(projectRoot, { commitHash: currentHash, files: fresh });
  return fresh;
}
```

```ts
import type { RankedFile } from './ranker.js';

export function renderRepoMap(files: RankedFile[], tokenBudget: number = 1024): string {
  const lines: string[] = [];
  let tokens = 0;
  for (const file of files) {
    lines.push(`${file.path}:`);
    for (const sym of file.symbols) {
      const line = `  ${sym.signature}`;
      tokens += Math.ceil(line.length / 4); // rough token estimate
      if (tokens > tokenBudget) return lines.join('\n');
      lines.push(line);
    }
  }
  return lines.join('\n');
}
```

```ts
import { generateText } from 'ai';
import { createProvider } from '@awecode/llm';
import type { AwecodeConfig } from '@awecode/llm';
import type { ContextEntry } from './manager.js';
import { countTokens } from 'gpt-tokenizer';

const SUMMARIZATION_PROMPT = `Summarize the conversation so far. PRESERVE:
1. Original user task statement
2. Key design decisions made
3. Files currently in context (paths + brief description)
4. Errors encountered and resolutions
5. Last 5 user-assistant turns (verbatim)

DISCARD:
- Verbose tool output (full file contents already in context entries)
- Redundant code reads
- Intermediate exploration that didn't lead to decisions

Output format: Markdown with sections [Task], [Decisions], [Files], [Errors], [Recent Turns].`;

export async function compactContext(
  config: AwecodeConfig,
  entries: ContextEntry[],
  recentTurns: Array<{ role: 'user' | 'assistant'; content: string }>,
): Promise<{ summary: string; tokensSaved: number }> {
  const providerConfig = config.providers[config.activeProvider];
  if (!providerConfig) throw new Error('No active provider');

  // Use small model if configured, else default
  const model = createProvider(providerConfig);
  const beforeTokens = entries.reduce((s, e) => s + e.tokens, 0);

  const conversationText = entries.map((e) => e.content).join('\n\n');
  const recentText = recentTurns.map((t) => `${t.role}: ${t.content}`).join('\n');

  const result = await generateText({
    model,
    system: SUMMARIZATION_PROMPT,
    prompt: `Conversation to summarize:\n\n${conversationText}\n\n--- Recent turns ---\n${recentText}`,
    maxTokens: 2048,
  });

  const afterTokens = countTokens(result.text);
  return {
    summary: result.text,
    tokensSaved: beforeTokens - afterTokens,
  };
}
```

```tsx
  const bar = '█'.repeat(filled) + '░'.repeat(20 - filled);

  return (
    <Box flexDirection="column">
      <Text>Context ({used.toLocaleString()} / {budget.toLocaleString()} tokens) — {pct}% — <Text color={color}>{level}</Text></Text>
      <Text color={color}>{bar}</Text>
    </Box>
  );
}
```
